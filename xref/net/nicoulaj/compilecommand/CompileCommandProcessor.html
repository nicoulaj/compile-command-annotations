50  /**
51   * {@link Processor} that generates a <code>hotspot_compiler</code> file for using with <code>-XX:CompileCommandFile</code>.
52   *
53   * @author <a href="http://github.com/nicoulaj">Julien Nicoulaud</a>
54   * @see <a href="http://docs.oracle.com/javase/8/docs/technotes/tools/unix/java.html"><code>CompileCommand</code>/<code>CompileCommandFile</code> documentation</a>
55   */
56  @MetaInfServices(Processor.class)
57  @SupportedSourceVersion(RELEASE_8)
58  @SupportedAnnotationTypes({"net.nicoulaj.compilecommand.annotations.Break",
59                             "net.nicoulaj.compilecommand.annotations.CompileOnly",
60                             "net.nicoulaj.compilecommand.annotations.DontInline",
61                             "net.nicoulaj.compilecommand.annotations.Exclude",
62                             "net.nicoulaj.compilecommand.annotations.Inline",
63                             "net.nicoulaj.compilecommand.annotations.Log",
64                             "net.nicoulaj.compilecommand.annotations.Option",
65                             "net.nicoulaj.compilecommand.annotations.Options",
66                             "net.nicoulaj.compilecommand.annotations.Print",
67                             "net.nicoulaj.compilecommand.annotations.Quiet"})
68  @SuppressWarnings("unused")
69  public final class CompileCommandProcessor extends AbstractProcessor {
70  
71      public static final String COMPILE_COMMAND_FILE_PATH_OPTION = "compile.command.file.output.path";
72  
73      public static final String COMPILE_COMMAND_FILE_PATH_DEFAULT = "META-INF/hotspot_compiler";
74  
75      private final SortedSet<String> lines;
76  
77      private boolean quiet = false;
78  
79      public CompileCommandProcessor() {
80          lines = new TreeSet<>();
81      }
82  
83      @Override
84      public boolean process(final Set<? extends TypeElement> annotations, final RoundEnvironment roundEnv) {
85  
86          info("Processing compiler hints annotations");
87  
88          processBreak(roundEnv);
89          processCompileOnly(roundEnv);
90          processDontInline(roundEnv);
91          processExclude(roundEnv);
92          processInline(roundEnv);
93          processLog(roundEnv);
94          processOptions(roundEnv);
95          processOption(roundEnv);
96          processPrint(roundEnv);
97          processQuiet(roundEnv);
98  
99          if (roundEnv.processingOver())
100             generateCompileCommandFile(processingEnv.getOptions().getOrDefault(COMPILE_COMMAND_FILE_PATH_OPTION, COMPILE_COMMAND_FILE_PATH_DEFAULT));
101 
102         info("Done processing compiler hints annotations");
103 
104         return true;
105     }
106 
107     private void processBreak(RoundEnvironment roundEnv) {
108         processSimpleMethodAnnotation(Break.class, roundEnv);
109     }
110 
111     private void processCompileOnly(RoundEnvironment roundEnv) {
112         processSimpleMethodAnnotation(CompileOnly.class, roundEnv);
113     }
114 
115     private void processDontInline(RoundEnvironment roundEnv) {
116         processSimpleMethodAnnotation(DontInline.class, roundEnv);
117     }
118 
119     private void processExclude(RoundEnvironment roundEnv) {
120         processSimpleMethodAnnotation(Exclude.class, roundEnv);
121     }
122 
123     private void processInline(RoundEnvironment roundEnv) {
124         processSimpleMethodAnnotation(Inline.class, roundEnv);
125     }
126 
127     private void processLog(RoundEnvironment roundEnv) {
128         processSimpleMethodAnnotation(Log.class, roundEnv);
129     }
130 
131     private void processOptions(RoundEnvironment roundEnv) {
132         for (Element element : roundEnv.getElementsAnnotatedWith(Options.class))
133             for (Option option : element.getAnnotation(Options.class).value())
134                 processOption(element, option, roundEnv);
135     }
136 
137     private void processOption(RoundEnvironment roundEnv) {
138         for (Element element : roundEnv.getElementsAnnotatedWith(Option.class))
139             processOption(element, element.getAnnotation(Option.class), roundEnv);
140     }
141 
142     private void processOption(Element element, Option option, RoundEnvironment roundEnv) {
143         lines.add(element.accept(new SimpleElementVisitor8<String, RoundEnvironment>() {
144 
145             @Override
146             public String visitExecutable(final ExecutableElement e, final RoundEnvironment roundEnvironment) {
147                 return Option.class.getSimpleName().toLowerCase() + " " + getDescriptor(e) + " " + option.value();
148             }
149 
150             @Override
151             protected String defaultAction(final Element e, final RoundEnvironment roundEnvironment) {
152                 error(e, "@%s is not allowed on a %s", Option.class.getSimpleName(), e.getKind());
153                 return null;
154             }
155         }, roundEnv));
156     }
157 
158     private void processPrint(RoundEnvironment roundEnv) {
159         processSimpleMethodAnnotation(Print.class, roundEnv);
160     }
161 
162     private void processQuiet(RoundEnvironment roundEnv) {
163         if (!roundEnv.getElementsAnnotatedWith(Quiet.class).isEmpty())
164             quiet = true;
165     }
166 
167     private void processSimpleMethodAnnotation(Class<? extends Annotation> clazz, RoundEnvironment roundEnv) {
168         for (Element element : roundEnv.getElementsAnnotatedWith(clazz))
169             lines.add(element.accept(new SimpleElementVisitor8<String, RoundEnvironment>() {
170 
171                 @Override
172                 public String visitExecutable(final ExecutableElement e, final RoundEnvironment roundEnvironment) {
173                     return clazz.getSimpleName().toLowerCase() + " " + getDescriptor(e);
174                 }
175 
176                 @Override
177                 protected String defaultAction(final Element e, final RoundEnvironment roundEnvironment) {
178                     error(e, "@%s is not allowed on a %s", clazz.getSimpleName(), e.getKind());
179                     return null;
180                 }
181             }, roundEnv));
182     }
183 
184     private String getDescriptor(ExecutableElement element) {
185         return processingEnv.getElementUtils().getBinaryName((TypeElement) element.getEnclosingElement())
186                + "::" + element.getSimpleName()
187                + " " + getSignature(element);
188     }
189 
190     private String getSignature(ExecutableElement element) {
191         final Types types = processingEnv.getTypeUtils();
192         final StringBuilder sb = new StringBuilder("(");
193         for (VariableElement p : element.getParameters())
194             sb.append(getSignature(types.erasure(p.asType())));
195         return sb.append(")")
196                  .append(getSignature(types.erasure(element.getReturnType())))
197                  .toString();
198     }
199 
200     private String getSignature(TypeMirror type) {
201         return type.accept(new SimpleTypeVisitor8<String, Void>() {
202 
203             @Override
204             public String visitPrimitive(final PrimitiveType t, final Void aVoid) {
205                 switch (t.toString()) {
206                 case "boolean":
207                     return "Z";
208                 case "short":
209                     return "S";
210                 case "int":
211                     return "I";
212                 case "long":
213                     return "J";
214                 case "float":
215                     return "F";
216                 case "double":
217                     return "D";
218                 case "char":
219                     return "C";
220                 case "byte":
221                     return "B";
222                 }
223                 return super.visitPrimitive(t, aVoid);
224             }
225 
226             @Override public String visitNoType(final NoType t, final Void aVoid) {
227                 return "V";
228             }
229 
230             @Override
231             public String visitArray(final ArrayType t, final Void aVoid) {
232                 return "[" + getSignature(t.getComponentType());
233             }
234 
235             @Override
236             protected String defaultAction(final TypeMirror e, final Void aVoid) {
237                 return "L" + e.toString() + ";";
238             }
239         }, null);
240     }
241 
242     private void generateCompileCommandFile(String path) {
243         try {
244             info("Writing compiler command file at %s", path);
245             final FileObject file = processingEnv.getFiler().createResource(CLASS_OUTPUT, "", path);
246             try (PrintWriter pw = new PrintWriter(new OutputStreamWriter(file.openOutputStream(), "UTF-8"))) {
247                 if (quiet) pw.println("quiet");
248                 for (String value : lines)
249                     pw.println(value);
250             }
251         } catch (IOException e) {
252             error("Failed writing compiler command file : %s", e);
253         }
254     }

257         message(NOTE, msg, args);
258     }
259

261         message(NOTE, element, msg, args);
262     }
263

265         message(WARNING, msg, args);
266     }
267 
268     private void warn(Element element, String msg, Object... args) {
269         message(WARNING, element, msg, args);
270     }
271 
272     private void error(String msg, Object... args) {
273         message(ERROR, msg, args);
274     }
275 
276     private void error(Element element, String msg, Object... args) {
277         message(ERROR, element, msg, args);
278     }
279 
280     private void message(Diagnostic.Kind level, String msg, Object... args) {
281         processingEnv.getMessager().printMessage(level, format(msg, args));
282     }
283 
284     private void message(Diagnostic.Kind level, Element element, String msg, Object... args) {
285         processingEnv.getMessager().printMessage(level, format(msg, args), element);
286     }
