23  import java.io.File;

25  import java.util.ArrayList;
26  import java.util.List;

28  import static com.google.common.collect.ObjectArrays.concat;
29  import static java.util.Arrays.sort;
30  import static net.nicoulaj.compilecommand.CompileCommandProcessor.COMPILE_COMMAND_FILE_PATH_DEFAULT;
31  import static net.nicoulaj.compilecommand.JavaCompilationTester.Report;
32  import static org.apache.commons.io.FilenameUtils.getBaseName;
33  import static org.apache.commons.io.FilenameUtils.getExtension;
34  import static org.fest.assertions.api.Assertions.assertThat;
35  import static org.testng.Assert.assertFalse;
36  import static org.testng.Assert.assertTrue;
37  
38  /**
39   * Unit tests for {@link CompileCommandProcessor}.
40   *
41   * @author <a href="http://github.com/nicoulaj">Julien Nicoulaud</a>
42   */
43  public final class CompileCommandProcessorTest {
44  
45      private static final JavaCompilationTester JAVAC = new JavaCompilationTester();
46  
47      private static final File SAMPLES_SOURCES = new File("src/samples/java/net/nicoulaj/compilecommand");
48  
49      private static final File SAMPLES_RESOURCES = new File("src/samples/resources/net/nicoulaj/compilecommand");
50  
51      private static final File TEST_CASES_SOURCES = new File("src/test/java/net/nicoulaj/compilecommand/testcases");
52  
53      private static final File TEST_CASES_RESOURCES = new File("src/test/resources/net/nicoulaj/compilecommand/testcases");
54  
55      @DataProvider
56      public Object[][] testcases() throws IOException {
57          return concat(getDataProvider(TEST_CASES_SOURCES, TEST_CASES_RESOURCES),
58                        getDataProvider(SAMPLES_SOURCES, SAMPLES_RESOURCES),
59                        Object[].class);
60      }
61  
62      private Object[][] getDataProvider(File sourceDir, File resourceDir) throws IOException {
63          final List<Object[]> data = new ArrayList<Object[]>();
64          final File[] sources = sourceDir.listFiles();
65          if (sources == null) throw new IllegalArgumentException("No source in " + sourceDir);
66          sort(sources);
67          for (File source : sources)
68              if ("java".equals(getExtension(source.toString())))
69                  if (!"package-info".equals(getBaseName(source.toString())))
70                      data.add(new Object[]{source, new File(resourceDir, getBaseName(source.toString()))});
71          return data.toArray(new Object[data.size()][]);
72      }
73  
74      @Test(dataProvider = "testcases")
75      public void test(File source, File expected) {
76          final Report compilation = JAVAC.compile(source);
77          assertTrue(compilation.isSuccessful(), "compilation failed");
78          assertFalse(compilation.hasErrors(), "compilation has errors");
79          assertFalse(compilation.hasWarnings(), "compilation has warnings");
80          assertThat(new File(compilation.getClassesDirectory(), COMPILE_COMMAND_FILE_PATH_DEFAULT)).hasContentEqualTo(expected);
81      }
82  }
