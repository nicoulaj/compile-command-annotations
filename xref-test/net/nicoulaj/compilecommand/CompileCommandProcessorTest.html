18  package net.nicoulaj.compilecommand;
19  
20  import org.testng.annotations.DataProvider;
21  import org.testng.annotations.Test;
22  
23  import java.io.IOException;
24  import java.nio.file.Files;
25  import java.nio.file.Path;
26  
27  import static java.nio.file.Paths.get;
28  import static net.nicoulaj.compilecommand.JavaCompilationTester.Report;
29  import static net.nicoulaj.compilecommand.CompileCommandProcessor.COMPILE_COMMAND_FILE_PATH_DEFAULT;
30  import static org.fest.assertions.api.Assertions.assertThat;
31  import static org.testng.Assert.assertFalse;
32  import static org.testng.Assert.assertTrue;
33  
34  /**
35   * Unit tests for {@link CompileCommandProcessor}.
36   *
37   * @author <a href="http://github.com/nicoulaj">Julien Nicoulaud</a>
38   */
39  public final class CompileCommandProcessorTest {
40  
41      private static final JavaCompilationTester JAVAC = new JavaCompilationTester();
42  
43      private static final Path SAMPLES_SOURCES = get("src/samples/java/net/nicoulaj/compilecommand");
44  
45      private static final Path SAMPLES_RESOURCES = get("src/samples/java/net/nicoulaj/compilecommand");
46  
47      private static final Path TEST_CASES_SOURCES = get("src/test/java/net/nicoulaj/compilecommand/testcases");
48  
49      private static final Path TEST_CASES_RESOURCES = get("src/test/resources/net/nicoulaj/compilecommand/testcases");
50  
51      @DataProvider
52      public Object[][] testcases() throws IOException {
53          return getDataProvider(TEST_CASES_SOURCES, TEST_CASES_RESOURCES);
54      }
55  
56      @DataProvider
57      public Object[][] samples() throws IOException {
58          return getDataProvider(SAMPLES_SOURCES, SAMPLES_RESOURCES);
59      }
60  
61      @Test(dataProvider = "testcases")
62      public void testCase(Path source, Path expected) {
63          test(source, expected);
64      }
65  
66      @Test(dataProvider = "samples")
67      public void testSample(Path source, Path expected) {
68          test(source, expected);
69      }
70  
71      private void test(Path source, Path expected) {
72          final Report compilation = JAVAC.compile(source);
73          assertTrue(compilation.isSuccessful(), "compilation failed");
74          assertFalse(compilation.hasErrors(), "compilation has errors");
75          assertFalse(compilation.hasWarnings(), "compilation has warnings");
76          assertThat(compilation.getClassesDirectory().resolve(COMPILE_COMMAND_FILE_PATH_DEFAULT).toFile()).hasContentEqualTo(expected.toFile());
77      }
78  
79      private Object[][] getDataProvider(Path sourceDir, Path resourceDir) throws IOException {
80          return Files.walk(sourceDir)
81                      .filter(path -> path.toString().endsWith(".java"))
82                      .filter(path -> !path.endsWith("package-info.java"))
83                      .sorted()
84                      .map(path -> new Object[]{path, resourceDir.resolve(path.getFileName().toString().replace(".java", ""))})
85                      .toArray(Object[][]::new);
86      }
87  }
