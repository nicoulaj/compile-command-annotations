18  package net.nicoulaj.compilecommand;
19  
20  import net.nicoulaj.compilecommand.annotations.DontInline;
21  import net.nicoulaj.compilecommand.annotations.Inline;
22  
23  public class CompileCommandAnnotations_01_Inlining {
24  
25      // This sample demonstrates how to force or prevent JIT from inlining a method.
26      //
27      // To run:
28      //   $ mvn clean test
29      //   $ java -cp target/test-classes \
30      //          -XX:+UnlockDiagnosticVMOptions \
31      //          -XX:+PrintInlining \
32      //          -XX:CompileCommandFile=src/samples/java/net/nicoulaj/compilecommand/CompileCommandAnnotations_01_Inlining \
33      //          net.nicoulaj.compilecommand.CompileCommandAnnotations_01_Inlining
34  
35      public static void main(String... args) {
36          for (int i = 0; i < 100_000; i++) {
37              method01();
38              method02();
39          }
40      }
41  
42      @Inline // Force the just in time compiler to attempt inlining the specified method.
43      private static void method01() {
44      }
45  
46      @DontInline // Prevent the just in time compiler from inlining the specified method.
47      private static void method02() {
48      }
49  }
